 FROM emp
 START WITH mgr IS NULL --루트노트
CONNECT BY PRIOR empno = mgr;

            PARENT_ID   INTEGER references BOM(item_id),
            ITEM_NAME   VARCHAR2(20)  NOT NULL,
            ITEM_QTY    INTEGER, 
            PRIMARY KEY (ITEM_ID)
);
